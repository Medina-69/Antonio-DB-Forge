-- Crear base de datos
CREATE DATABASE [Medina_ERP];
GO
USE [Medina_ERP];
GO

-- Tabla Proveedores
CREATE TABLE Proveedores (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Nombre VARCHAR(50),
    Prefijo VARCHAR(2),
    RFC VARCHAR(10) CHECK (RFC = UPPER(RFC)),
    Homoclave VARCHAR(4),
    Direccion VARCHAR(150),
    Telefono VARCHAR(20),
    Email VARCHAR(100) CHECK (Email LIKE '%@%')
);

-- Tabla Marcas
CREATE TABLE Marcas (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Marca VARCHAR(50),
    Imagen VARBINARY(MAX)  -- Para almacenar archivos de imagen
);

-- Tabla Productos
CREATE TABLE Productos (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Codigo VARCHAR(30),
    Nombre VARCHAR(150),
    Descripcion VARCHAR(450),
    Id_Marca INT FOREIGN KEY REFERENCES Marcas(Id),
    Costo_Unitario FLOAT,
    Unidad VARCHAR(2),
    Imagen_Principal VARBINARY(MAX),
    Imagen_1 VARBINARY(MAX),
    Imagen_2 VARBINARY(MAX),
    Imagen_3 VARBINARY(MAX)
);

-- Tabla OrdenesCompra
CREATE TABLE OrdenesCompra (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Fecha DATE,
    No_Orden VARCHAR(15),
    Id_Proveedor INT FOREIGN KEY REFERENCES Proveedores(Id),
    Id_Producto INT FOREIGN KEY REFERENCES Productos(Id),
    Costo_Unitario FLOAT,
    Descuento FLOAT CHECK (Descuento BETWEEN 0 AND 100),
    Cantidad INT
);

-- Tabla Compras
CREATE TABLE Compras (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    No_Orden VARCHAR(15),
    SubTotal FLOAT,
    IVA FLOAT,
    Total FLOAT,
    Aceptado DATE,
    [Stand-by] DATE,
    Pagado DATE,
    Cancelado DATE
);

-- Tabla Clientes
CREATE TABLE Clientes (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Nombre_Completo VARCHAR(500),
    Apellido_Paterno VARCHAR(150),
    Apellido_Materno VARCHAR(150),
    Nombres VARCHAR(150),
    RFC VARCHAR(10),
    Homoclave VARCHAR(4),
    Direccion VARCHAR(50),
    Ciudad VARCHAR(50),
    Estado VARCHAR(50),
    CP INT,
    Telefono VARCHAR(20),
    Email VARCHAR(50) CHECK (Email LIKE '%@%')
);

-- Tabla OrdenVenta
CREATE TABLE OrdenVenta (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Fecha DATE,
    No_Orden VARCHAR(15),
    Id_Cliente INT FOREIGN KEY REFERENCES Clientes(Id),
    Id_Producto INT FOREIGN KEY REFERENCES Productos(Id),
    Cantidad INT,
    Descuento FLOAT CHECK (Descuento BETWEEN 0 AND 100)
);

-- Tabla Ventas
CREATE TABLE Ventas (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    No_Orden VARCHAR(15),
    SubTotal FLOAT,
    IVA FLOAT,
    Total FLOAT
);
